CREATE PROCEDURE sp_KiemTraChiTietPhieuNhap
@MaPN nchar(8),
@MaVT nchar(4)
AS
BEGIN
	IF(EXISTS(SELECT 1 FROM CTPN AS P WHERE P.MAPN = @MaPN AND P.MAVT = @MaVT))
		RETURN 1;
	ELSE IF( EXISTS(SELECT 1 FROM LINK0.QLVT_DATHANG.DBO.CTPN AS P WHERE P.MAPN = @MaPN AND P.MAVT = @MaVT))
		RETURN 1;
	RETURN 0;
END
